/** * build_channel.gs — витрина «Instagram-канал» по месяцам * ------------------------------------------------ * Сопоставляет ПОМЕСЯЧНО: * расход Instagram-рекламы (из Meta, все кампании всех кабинетов) * × количество IG-заявок из amoCRM (source = Instagram) * * Данные: * - расход берётся напрямую из Meta через getMetaMonthly_() (тот же токен/аккаунты) * - заявки — из листа RAW_leads (наполняет etl_amo.gs) * * Уровень: КАНАЛ Instagram целиком, НЕ по кампаниям (в данных amo нет привязки к кампании). * Выручки в деньгах нет (в amoCRM price=0; деньги в Alpha, подключим позже). * * Script Properties: SHEET_ID, FB_TOKEN, AD_ACCOUNTS (уже заданы) * Запуск: buildChannel() — затем данные читает getChannelJson() через ?view=channel */ var SH_LEADS_CH = 'RAW_leads'; var SH_CHANNEL = 'MART_channel_month'; var FB_API_VER_CH = 'v25.0'; function buildChannel() { var props = PropertiesService.getScriptProperties(); var sheetId = props.getProperty('SHEET_ID'); if (!sheetId) throw new Error('Не задан SHEET_ID'); var ss = SpreadsheetApp.openById(sheetId); // 1) расход IG по месяцам из Meta (за последние 13 месяцев) var spendByMonth = getMetaMonthly_(props, 13); // 2) IG-заявки по месяцам из RAW_leads var sh = ss.getSheetByName(SH_LEADS_CH); if (!sh) throw new Error('Нет листа ' + SH_LEADS_CH); var data = sh.getDataRange().getValues(); var header = data.shift(); var iDate = header.indexOf('created_at'); var iSource = header.indexOf('source'); var iStage = header.indexOf('stage'); var leadsByMonth = {}; // 'YYYY-MM' -> {leads, won} data.forEach(function (r) { if (!/^(instagram|ig)$/i.test(String(r[iSource] || '').trim())) return; var d = r[iDate]; if (!(d instanceof Date)) return; var m = d.getFullYear() + '-' + ('0' + (d.getMonth() + 1)).slice(-2); if (!leadsByMonth[m]) leadsByMonth[m] = { leads: 0, won: 0 }; leadsByMonth[m].leads++; if (/успешно реализовано/i.test(String(r[iStage] || ''))) leadsByMonth[m].won++; }); // 3) объединяем по месяцам var months = {}; Object.keys(spendByMonth).forEach(function (m) { months[m] = true; }); Object.keys(leadsByMonth).forEach(function (m) { months[m] = true; }); var rows = Object.keys(months).sort().map(function (m) { var sp = spendByMonth[m] || 0; var lb = leadsByMonth[m] || { leads: 0, won: 0 }; var cpl = lb.leads ? Math.round(sp / lb.leads * 100) / 100 : ''; return [m, sp, lb.leads, lb.won, cpl]; }); var sh2 = writeSheet_(ss, SH_CHANNEL, ['month', 'ig_spend_usd', 'ig_leads', 'ig_won', 'cost_per_lead_usd'], rows); if (rows.length) sh2.getRange(2, 1, rows.length, 1).setNumberFormat('@'); // month как текст Logger.log('MART_channel_month: ' + rows.length + ' месяцев'); } /** Расход Instagram по месяцам из Meta по всем аккаунтам из AD_ACCOUNTS */ function getMetaMonthly_(props, monthsBack) { var token = props.getProperty('FB_TOKEN'); var raw = props.getProperty('AD_ACCOUNTS') || props.getProperty('AD_ACCOUNT_ID') || ''; if (!token || !raw) throw new Error('Нет FB_TOKEN / AD_ACCOUNTS'); var accounts = raw.split(',').map(function (p) { return p.trim().split(':')[0].replace('act_', '').trim(); }).filter(Boolean); var until = new Date(); var since = new Date(until.getFullYear(), until.getMonth() - monthsBack, 1); var tr = encodeURIComponent(JSON.stringify({ since: Utilities.formatDate(since, 'Europe/Minsk', 'yyyy-MM-dd'), until: Utilities.formatDate(until, 'Europe/Minsk', 'yyyy-MM-dd') })); var byMonth = {}; accounts.forEach(function (id) { // time_increment=monthly недоступен напрямую — берём по дням и агрегируем var url = 'https://graph.facebook.com/' + FB_API_VER_CH + '/act_' + id + '/insights' + '?level=account&time_increment=1&time_range=' + tr + '&fields=spend&limit=500&access_token=' + encodeURIComponent(token); while (url) { var resp = UrlFetchApp.fetch(url, { muteHttpExceptions: true }); var d = JSON.parse(resp.getContentText()); if (d.error) break; // молча пропускаем недоступный кабинет (d.data || []).forEach(function (row) { var m = row.date_start.slice(0, 7); byMonth[m] = (byMonth[m] || 0) + Number(row.spend || 0); }); url = (d.paging && d.paging.next) ? d.paging.next : null; } }); // округление Object.keys(byMonth).forEach(function (m) { byMonth[m] = Math.round(byMonth[m] * 100) / 100; }); return byMonth; } function writeSheet_(ss, name, header, rows) { var sh = ss.getSheetByName(name) || ss.insertSheet(name); sh.clearContents(); sh.getRange(1, 1, 1, header.length).setValues([header]); if (rows.length) sh.getRange(2, 1, rows.length, header.length).setValues(rows); return sh; } /** JSON для дашборда: ?view=channel */ function getChannelJson() { var sheetId = PropertiesService.getScriptProperties().getProperty('SHEET_ID'); var ss = SpreadsheetApp.openById(sheetId); var sh = ss.getSheetByName(SH_CHANNEL); if (!sh) return { months: [] }; var d = sh.getDataRange().getValues(); d.shift(); return { updated: new Date().toISOString(), months: d.map(function (r) { return { month: r[0], spend: r[1], leads: r[2], won: r[3], cpl: r[4] }; }) }; }
